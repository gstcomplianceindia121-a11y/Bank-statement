# extractor.py: Smart extraction (Auto-detects Text/Image).
import pdfplumber
from pathlib import Path
import logging

logger = logging.getLogger(__name__)

def _is_scanned_pdf(pdf_path: Path) -> bool:
    with pdfplumber.open(pdf_path) as pdf:
        for page in pdf.pages:
            if page.chars: return False
    return True

def extract(pdf_path: str | Path) -> list[str]:
    pdf_path = Path(pdf_path)
    if _is_scanned_pdf(pdf_path):
        logger.info("🔎 Scanned/Image PDF detected. Switching to OCR...")
        return extract_ocr_pdf(pdf_path) # Uses PaddleOCR
    
    logger.info("📑 Text-based PDF detected. Using fast extraction...")
    all_lines = []
    with pdfplumber.open(pdf_path) as pdf:
        for page in pdf.pages:
            text = page.extract_text(x_tolerance=3, y_tolerance=3) or ""
            all_lines.extend(text.splitlines())
    return all_lines

# llm_parser.py: Gemini 2.5 Semantic Engine (Schema-Controlled & Context-Aware)
from google import genai
import json, os, logging

logger = logging.getLogger(__name__)

def parse_transactions_llm(lines: list[str]) -> dict:
    client = genai.Client(api_key=os.getenv("GOOGLE_API_KEY"))
    
    all_transactions = []
    final_metadata = {"account_info": {}, "summary": {}}
    context_tail = ""
    batch_size = 70  # Roughly 1 full page

    for i in range(0, len(lines), batch_size):
        current_chunk = lines[i : i + batch_size]
        text_to_parse = "\n".join(current_chunk)
        
        # Build continuity prompt
        prompt = f"CONTINUITY CONTEXT:\n{context_tail}\n\nCURRENT PAGE:\n{text_to_parse}\n"
        prompt += "\nReturn JSON with schema: {account_info, transactions, summary}"
        
        response = client.models.generate_content(
            model="gemini-2.0-flash",
            contents=prompt,
            config={'response_mime_type': 'application/json'}
        )
        data = json.loads(response.text.strip())
        
        # Merge results
        if data.get("transactions"): all_transactions.extend(data["transactions"])
        if not final_metadata["account_info"] and "account_info" in data:
            final_metadata["account_info"] = data["account_info"]

        # Update context for next page
        context_tail = "\n".join(current_chunk[-3:])

    return { "account_info": final_metadata["account_info"], "transactions": all_transactions }

# validator.py: Advanced Math Verification
def validate_transactions(records: list[dict], summary_stats: dict = None):
    issues = []
    prev_bal = None
    for idx, rec in enumerate(records):
        debit, credit = rec.get("debit", 0.0), rec.get("credit", 0.0)
        bal = rec.get("balance")
        if prev_bal is not None:
            expected = round(prev_bal - debit + credit, 2)
            if abs(bal - expected) > 0.05:
                rec["validation_message"] = "FLAGGED: Balance Mismatch"
                issues.append({"level": "WARNING", "location": f"Row {idx+1}", "message": "Math Mismatch"})
        prev_bal = bal
    return records, issues

# cleaner.py: Standardizes dates and strips commas from amounts.
import re
from datetime import datetime

def clean_transactions(raw_records: list[dict]):
    cleaned = []
    for rec in raw_records:
        rec["date"] = datetime.strptime(rec["date"], "%d/%m/%Y").strftime("%d-%m-%Y")
        rec["debit"] = float(str(rec["debit"]).replace(",", ""))
        cleaned.append(rec)
    return cleaned

# categorizer.py: Maps descriptions to Tally Voucher Types.
def categorize_transactions(records: list[dict]):
    for rec in records:
        desc = rec.get("description", "").upper()
        if "ATM" in desc: rec["vch_type"] = "Cash Withdrawal"
        elif "UPI" in desc: rec["vch_type"] = "UPI Transfer"
        else: rec["vch_type"] = "General"
    return records
